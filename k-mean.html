kmeans = KMeans(n_clusters=7, init="random",n_init=10, max_iter=500, tol=1e-4, random_state=0).fit(X)
labels = kmeans.labels_
plt.figure(figsize=(14,14))
plt.scatter(X[labels==0, 0], X[labels==0, 1], s=20, c='red', label ='Cluster 1',marker="o")
plt.scatter(X[labels==1, 0], X[labels==1, 1], s=20, c='blue', label ='Cluster 2',marker="o")
plt.scatter(X[labels==2, 0], X[labels==2, 1], s=20, c='aqua', label ='Cluster 3',marker="o")
